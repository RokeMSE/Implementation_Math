from fractions import Fraction
import time

class Matrix:
     def __init__(self, A):
          self.A = A
          self.n = len(A)
          self.m = len(A[0])
          
     def if_square(self):
          # Kim tra ma trn vung
          if len(self.A) == len(self.A[0]):
               return 1
          return 0
     
     def swap_row(self, i, j):
          self.A[i], self.A[j] = self.A[j], self.A[i] 
          
     def add_row(self, i, j, k):
          for l in range(self.m):
               self.A[i][l] += k*self.A[j][l] # Cng hng j nhn vi k vo hng i
               
     def multiply_row(self, i, k):
          for l in range(self.m):
               self.A[i][l] *= k # Nhn hng i vi k

     def inverse(self):
          start_time = time.time()
          
          if not (self.if_square()):
               print("Ma trn khng kh nghch v khng vung")
               print()
               return None
          
          # To ma trn n v
          I = [[1 if (i==j) else 0 for i in range(self.n)] for j in range(self.n)]
          unit_matrix = Matrix(I)
          
          # To ma trn b sung
          # Format: (A|I)
          B = []
          for i in range(self.n):
               B.append(self.A[i] + unit_matrix.A[i])
          B = Matrix(B)

          # p Gauss-Jordan. Cc bc thc hin
          # B1: Tm hng c phn t khc 0  ct i
          # B2: Chia hng i cho phn t  ct i
          # B3: Loi b cc phn t  ct i trong cc hng khc (khc Gauss Elimination  ch hng trn v di cng phi loi b phn t  ct i)
          for i in range(self.n):
               # Tm hng c phn t khc 0  ct i
               for j in range(i, self.n):
                    if B.A[j][i] != 0:
                         B.swap_row(i, j)
                         break
               # Chia hng i cho phn t  ct i
               if B.A[i][i] != 0:
                    B.multiply_row(i, Fraction(1,B.A[i][i]))
               else:
                    print("Ma trn khng kh nghch") # V dng i c phn t  ct i bng 0
                    print()
                    return None
               # Loi b cc phn t  ct i trong cc hng khc
               for j in range(self.n):
                    if j != i:
                         B.add_row(j, i, -B.A[j][i])
          
          end_time = time.time()
          print("Manual: ", end_time - start_time)
          return B 
     
     def print_matrix(self):
          for i in range(self.n):
               for j in range(self.m):
                    print(self.A[i][j], end = " ")
               print()
          
def print_combine_matrix(A):
     # V ma trn lun vung nn mnh s chia ma trn ra lm 2 phn c 
     for i in range(A.n):
          for j in range(A.m//2): # In phn A
               print(A.A[i][j], end = " ")
          print("|", end = " ")
          for j in range(A.m//2, A.m): # In phn I
               print(A.A[i][j], end = " ")
          print()
     print()

def solveWithLib(A):

    print(np.linalg.inv(A))
    end_time = time.time()
    numpy_time = end_time - start_time
    print(f"NumPy: {numpy_time}s")
    print()

    start_time = time.time()
    A = sp.Matrix(A)

    end_time = time.time()
